const express = require('express'); const { handleMessage } = require('./game'); const app = express(); app.use(express.json()); // مسار الاستقبال لعرض الصفحة وفي نفس الوقت التقاط إشعارات الماكرو app.get('/', (req, res) => { // استخراج بيانات الإشعار المرسلة من MacroDroid const notifTitle = req.query.title || "بدون عنوان"; const notifText = req.query.text || "بدون نص"; const notifApp = req.query.app || "تطبيق غير معروف"; // طباعة تفاصيل الإشعار بوضوح في سجلات Render console.log(`🔔 إشعار من [${notifApp}] | العنوان: ${notifTitle} | النص: ${notifText}`); res.send(` البوت شغال بنجاح

🤖 بوت Gem Space شغال

السيرفر يعمل 24 ساعة وجاهز لاستقبال الأوامر بنجاح!

`); }); app.post('/webhook', (req, res) => { const message = req.body.message || req.body.text || ""; const sender = req.body.senderId || "user"; const reply = handleMessage(message, sender); if (reply) { return res.json({ reply: reply }); } res.status(200).send('OK'); }); const PORT = process.env.PORT || 10000; app.listen(PORT, () => { console.log(`Server running on port ${PORT}`); });
